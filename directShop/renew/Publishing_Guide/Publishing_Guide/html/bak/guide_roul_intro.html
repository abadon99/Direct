css/js 폴더 구조

								// css folder
								/inc/css/import.css                  // css 가져오기
								/inc/css/ini.min.css                 // 스타일 초기화
								/inc/css/common.css                  // 공통사항
								/inc/css/layout.css                  // 레이아웃
								/inc/css/main.css                    // 메인페이지
								/inc/css/content.css                 // 콘텐츠(서브)
								/inc/css/plugins.css                 // 플러그인

								// js folder
								/inc/js/libs/jquery.min.js           // jQuery core
								/inc/js/libs/html5shiv.js            // html5 호환성
								/inc/js/libs/기타 플러그인 파일      // 외부 플러그인

								/inc/js/pub/main.js                  // 메인 UI
								/inc/js/pub/ui.js                    // 공통 UI

								// image folder
								/inc/img/common                  // 공통이미지

								/inc/img/sub                     // 서브페이지
